    This data consists of temperatures of INDIA averaging the temperatures of all places month wise.
    Temperatures values are recorded in CELSIUS temperature. 
    b. Assess the performance of regression models using MSE, MAE and R-Square metrics 
    c. Visualize simple regression model

# Import the pandas library for data manipulation and analysis, assigning it the alias 'pd'.
import pandas as pd
# Import the train_test_split function to divide the dataset into training and testing subsets.
from sklearn.model_selection import train_test_split
# Import the pyplot module from matplotlib for creating plots and visualizations, assigning it the alias 'plt'.
import matplotlib.pyplot as plt
# Import the metrics module from scikit-learn, which contains functions for evaluating model performance.
from sklearn import metrics
# Import the LinearRegression class, which is the model we will use for prediction.
from sklearn.linear_model import LinearRegression

# Load the data from the 'weight-height.csv' file into a pandas DataFrame named 'df'.
df= pd.read_csv('weight-height.csv')
# Display the entire DataFrame to examine the data's initial structure and content.
df
# Remove the 'Gender' column from the DataFrame, as we are only using Height to predict Weight (setting axis="columns").
df=df.drop("Gender", axis="columns")

# Intended to show a summary of the DataFrame (data types, non-null counts), but is missing parentheses '()'.
df.info
# Display the dimensions of the DataFrame (number of rows and columns).
df.shape
# Check for and sum the total number of missing values (NaN) in each column.
df.isna().sum()
# Check for and sum the total number of fully duplicated rows in the DataFrame.
df.duplicated().sum()

# Select the first column (Height) as the independent variable (feature) 'x', keeping it in DataFrame format.
x=df.iloc[:,0:1]
# Select the second column (Weight) as the dependent variable (target) 'y', keeping it in DataFrame format.
y=df.iloc[:,1:2]

# Split the data into training (80%) and testing (20%) sets for both x and y.
x_train,x_test,y_train,y_test=train_test_split(x, y, test_size=0.2)
# Create an instance (object) of the Linear Regression model.
Reg=LinearRegression()
# Train the model using the training data (x_train to predict y_train).
Reg.fit(x_train,y_train)
# Use the trained model to make weight predictions (y_predict) on the test heights (x_test).
y_predict=Reg.predict(x_test)
# Display the array of predicted weight values.
y_predict

# Print the coefficient (slope) of the fitted line.
print(Reg.coef_)
# Print the intercept (y-axis crossing point) of the fitted line.
print(Reg.intercept_)

# Create a scatter plot of the actual test data (x_test vs. y_test).
plt.scatter(x_test,y_test)
# Overlay the regression line onto the scatter plot using the predicted values (y_predict), colored red.
plt.plot(x_test,y_predict,color="red")

# Calculate and print the Mean Squared Error (MSE) of the model's predictions on the test data.
print('meansquarerror', metrics.mean_squared_error(y_test,y_predict))
# Calculate and print the Mean Absolute Error (MAE) of the model's predictions on the test data.
print('meanabsoluteerror', metrics.mean_absolute_error(y_test,y_predict))
# Calculate the R-squared score (R2) using the training data and store it in the 'Rsquare' variable.
Rsquare=Reg.score(x_train,y_train)
# Print the calculated R-squared score.
print (Rsquare)
